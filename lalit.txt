SELECT  CL.PHASE_NAME,CL.COUNTRY_DESC, CL.QUESTION_DETAILS, CL.CHKLIST_FLAG AS central_lab, IRT.CHKLIST_FLAG as irt, ECG.CHKLIST_FLAG as ecg, CL.comments
FROM  (
SELECT  PQD.PHASEWISE_QUESTION_ID,c.country_desc,PCD.CHKLIST_FLAG,PCD.VENDOR_CATEGORY_ID, QM.QUESTION_DETAILS, PQD.comments,PQD.PHASE_NAME
FROM 
PHASEWISE_QUESTION_DTLS PQD
,PHASEWISE_CHKLIST_DTLS PCD,QUESTION_MASTER QM
,REGION_COUNTRY_MAPPING RCM
,COUNTRY C
WHERE PQD.PHASEWISE_QUESTION_ID in (:question_id)
AND PQD.PHASEWISE_QUESTION_ID=PCD.PHASEWISE_QUESTION_ID
aND PQD.PHASE_NAME IN (:phase_name)
AND QM.QUESTION_ID=PQD.QUESTION_ID
AND RCM.ID in (:region_id)
AND RCM.ID=PQD.REGION_COUNTRY_ID
AND C.COUNTRY_ID in (:country_id)
AND C.COUNTRY_ID=RCM.COUNTRY_ID
and PCD.VENDOR_CATEGORY_ID = 1
)   CL,

(
SELECT PQD.PHASEWISE_QUESTION_ID,c.country_desc,PCD.CHKLIST_FLAG,PCD.VENDOR_CATEGORY_ID, QM.QUESTION_DETAILS, PQD.comments,PQD.PHASE_NAME
FROM 
PHASEWISE_QUESTION_DTLS PQD
,PHASEWISE_CHKLIST_DTLS PCD,QUESTION_MASTER QM
,REGION_COUNTRY_MAPPING RCM
,COUNTRY C
WHERE PQD.PHASEWISE_QUESTION_ID in (:question_id)
AND PQD.PHASEWISE_QUESTION_ID=PCD.PHASEWISE_QUESTION_ID
aND PQD.PHASE_NAME IN (:phase_name)
AND QM.QUESTION_ID=PQD.QUESTION_ID
AND RCM.ID in (:region_id)
AND RCM.ID=PQD.REGION_COUNTRY_ID
AND C.COUNTRY_ID in (:country_id)
AND C.COUNTRY_ID=RCM.COUNTRY_ID
and PCD.VENDOR_CATEGORY_ID = 2
)   IRT
,
(
SELECT  PQD.PHASEWISE_QUESTION_ID,c.country_desc,PCD.CHKLIST_FLAG,PCD.VENDOR_CATEGORY_ID, QM.QUESTION_DETAILS,PQD.comments,PQD.PHASE_NAME
FROM 
PHASEWISE_QUESTION_DTLS PQD
,PHASEWISE_CHKLIST_DTLS PCD,QUESTION_MASTER QM
,REGION_COUNTRY_MAPPING RCM
,COUNTRY C
WHERE PQD.PHASEWISE_QUESTION_ID in (:question_id)
AND PQD.PHASEWISE_QUESTION_ID=PCD.PHASEWISE_QUESTION_ID
aND PQD.PHASE_NAME IN (:phase_name)
AND QM.QUESTION_ID=PQD.QUESTION_ID
AND RCM.ID in (:region_id)
AND RCM.ID=PQD.REGION_COUNTRY_ID
AND C.COUNTRY_ID in (:country_id)
AND C.COUNTRY_ID=RCM.COUNTRY_ID
)   ECG 

where CL.VENDOR_CATEGORY_ID = 1
and IRT.VENDOR_CATEGORY_ID = 2
and ECG.VENDOR_CATEGORY_ID = 3
and CL.PHASEWISE_QUESTION_ID = IRT.PHASEWISE_QUESTION_ID
and IRT.PHASEWISE_QUESTION_ID = ECG.PHASEWISE_QUESTION_ID;
